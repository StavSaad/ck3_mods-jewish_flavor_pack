language_aramaic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_aramaic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_aramaic }
			multiply = 10
		}
	}
	
	color = assyrian
}

language_israelite = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_israelite
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_israelite }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					religion = religion:judaism_religion
				}
			}
			multiply = 2
		}
	}
	
	color = hebrew
}

language_canaanite = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_canaanite
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_canaanite }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					AND = {
						NOT = {
							religion = religion:judaism_religion
						}
						culture = {
							has_cultural_pillar = heritage_israelite
						}
					}
				}
			}
			multiply = 2
		}
	}
	
	color = canaanite
}

language_punic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_punic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_punic }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					AND = {
						NOT = {
							religion = religion:judaism_religion
						}
						culture = {
							OR = {
								has_cultural_pillar = heritage_israelite
								any_parent_culture = { has_cultural_pillar = language_canaanite }
							}
						}
					}
				}
			}
			multiply = 2
		}
	}
	
	color = bavlim
}
